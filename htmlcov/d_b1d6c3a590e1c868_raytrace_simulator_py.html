110def traces_rays(ray_position, rays, planes, bounces, out): 

111 """ 

112 

113 :param ray_position: 

114 :param rays: 

115 :param planes: group plane data 

116 :param bounces: 

117 :param out: 

118 :return: [plane_id, x, y, z] x,y,z point of intersection. if plane_id = -1 then no intersection. 

119 """ 

120 

121 # Loop through each ray until it reaches max bounces or absorbs into surface. 

122 for i in range(rays.shape[0]): 

123 ray = rays[i, :] 

124 ray_position_now = ray_position 

125 bounces_count = 0 

126 skip_plane = -1 

127 for _ in range(bounces + 1): 

128 bounces_count += 1 

129 

130 # calculate plane intersections 

131 for plane_id, plane in enumerate(planes): 

132 if plane_id != skip_plane: 

133 # check to see if ray will hit infinite plane 

134 intersect_cord = plane_ray_intersection(rays_dir=ray, 

135 rays_pos=ray_position_now, 

136 plane_dir=plane[7:10], 

137 plane_pos=plane[4:7]) 

138 if intersect_cord is not None: 

139 # check to see if the hit is within the bounds of the plane 

140 if check_in_plane_range(point=intersect_cord, plane_corners=plane[10:]): 

141 skip_plane = plane_id 

142 out[i, 0] = plane_id 

143 out[i, 1:] = intersect_cord 

144 break 

145 

146 if bounces_count <= bounces: # skip if no bounces left 

147 # calculate angle light hit plane 

148 angle = np.arcsin(np.dot(ray, plane[7:10])) 

149 # transmitted light 

150 if 0 < plane[0] <= 2: 

151 # calculate probably of light transmitting given the angle. 

152 prob = lens_details.function_selector(plane[1], np.array([angle], dtype="float64")) 

153 if np.random.random() < prob: 

154 # if transmitted, calculate diffraction ray and continue tracing the ray 

155 ray = create_rays(theta_fun_id=plane[2], num_rays=1, direction=ray) 

156 ray = np.reshape(ray, 3) 

157 ray_position_now = out[i, 1:] 

158 continue

162 # calculate probably of light transmitting given the angle.
